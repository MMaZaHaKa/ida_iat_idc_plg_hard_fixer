#include <idc.idc>

static GetMZBase() {
    auto base = SegStart(MinEA());
    return base - 0x1000;
}

static GetMZBaseFunc() {
    auto base = GetMZBase();
    return base;
}

static find_tab(s, offset)
{
    auto p;
    p = strstr(substr(s, offset, -1), "\t");
    if (p == -1)
        return -1;
    return offset + p;
}

static get_field(s, idx)
{
    auto i, start, tab;

    start = 0;
    for (i = 0; i < idx; i++)
    {
        tab = find_tab(s, start);
        if (tab == -1)
            return "";
        start = tab + 1;
    }

    tab = find_tab(s, start);
    if (tab == -1)
        tab = strlen(s);

    return substr(s, start, tab);
}

static ReadTxtLine(fp)
{
    auto line, ch;

    line = "";
    ch = fgetc(fp);
    if (ch == -1)
        return -1;

    while (ch != -1 && ch != 10 && ch != 13)
    {
        line = line + sprintf("%c", ch);
        ch = fgetc(fp);
    }

    if (ch == 13)
    {
        ch = fgetc(fp);
        if (ch != 10 && ch != -1)
            fseek(fp, -1, 1);
    }

    return line;
}

static main()
{
    auto base, file, fp, line;
    auto rva_s, name, address;

    base = GetMZBase();

    file = AskFile(0, "*.*", "Select file with offsets");
    if (file == 0)
        return;

    fp = fopen(file, "r");
    if (fp == 0)
        return;

    while (1)
    {
        line = ReadTxtLine(fp);
        if (line == -1)
            break;

        if (strlen(line) == 0)
            continue;

        // 3-щ ёҐюысхІ = RVA, 5-щ ёҐюысхІ = name
        rva_s = get_field(line, 2);
        name  = get_field(line, 4);

        if (rva_s == "" || name == "")
            continue;

        address = base + xtol(rva_s);

        MakeCode(address);
        add_func(address, BADADDR);
        MakeName(address, name);
        SetColor(address, CIC_FUNC, 0x005500);

        Message("Added: %s at %a\n", name, address);
    }

    fclose(fp);
    Message("Done!\n");
}
